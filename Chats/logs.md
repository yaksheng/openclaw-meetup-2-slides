def extract_arxiv_id(identifier):
    """Extract arXiv ID from URL or raw ID."""
    if identifier.startswith('http'):
        # URL format: https://arxiv.org/abs/2403.01234
        match = re.search(r'arxiv\.org/(?:abs|pdf)/(\d+\.\d+)', identifier)
        if match:
            return match.group(1)
    else:
        # Raw ID format: 2403.01234
        match = re.match(r'(\d+\.\d+)', identifier)
        if match:
            return match.group(1)
    return None


def fetch_arxiv_metadata(arxiv_id):
    """Fetch paper metadata from arXiv API."""
    api_url = f"http://export.arxiv.org/api/query?id_list={arxiv_id}"
    
    try:
        with urllib.request.urlopen(api_url, timeout=30) as response:
            xml_content = response.read().decode('utf-8')
        
        # Parse XML manually (avoiding lxml dependency)
        data = {
            'arxiv_id': arxiv_id,
            'title': '',
            'authors': [],
            'abstract': '',
            'published': '',
            'categories': [],
            'pdf_url': f'https://arxiv.org/pdf/{arxiv_id}.pdf',
            'abs_url': f'https://arxiv.org/abs/{arxiv_id}'
        }
        
        # Extract title
        title_match = re.search(r'<title>(.*?)</title>', xml_content, re.DOTALL)
        if title_match:
            data['title'] = title_match.group(1).strip().replace('\n', ' ')
        
        # Extract authors
        authors = re.findall(r'<name>(.*?)</name>', xml_content)
        data['authors'] = [a.strip() for a in authors]
        
        # Extract abstract
        abstract_match = re.search(r'<summary>(.*?)</summary>', xml_content, re.DOTALL)
        if abstract_match:
            data['abstract'] = abstract_match.group(1).strip().replace('\n', ' ')
        
        # Extract published date
        pub_match = re.search(r'<published>(.*?)</published>', xml_content)
        if pub_match:
            data['published'] = pub_match.group(1)
        
        # Extract categories
        categories = re.findall(r'<term>(.*?)</term>', xml_content)
        data['categories'] = list(set(c.strip() for c in categories))
        
        return data
        
    except Exception as e:
        return {'error': str(e), 'arxiv_id': arxiv_id}


def save_paper_data(data, output_dir):
    """Save paper metadata to JSON file."""
    os.makedirs(output_dir, exist_ok=True)
    output_file = os.path.join(output_dir, f"{data['arxiv_id']}.json")
    
    with open(output_file, 'w', encoding='utf-8') as f:
        json.dump(data, f, indent=2, ensure_ascii=False)
    
    return output_file
